    /// Test IDF score calculation and weight assignment
    #[test]
    fn test_idf_score_and_weight_calculation() {
        let total_functions = 1000;
        
        // High frequency pattern (appears in 80% of functions)
        let high_freq_support = 800;
        let high_freq_idf = ((1.0 + total_functions as f64) / (1.0 + high_freq_support as f64)).ln() + 1.0;
        
        // Low frequency pattern (appears in 5% of functions)
        let low_freq_support = 50;
        let low_freq_idf = ((1.0 + total_functions as f64) / (1.0 + low_freq_support as f64)).ln() + 1.0;

        assert!(low_freq_idf &gt; high_freq_idf,
                &quot;Low frequency patterns should have higher IDF scores: {} vs {}&quot;, 
                low_freq_idf, high_freq_idf);

        // Test weight multiplier assignment based on frequency
        let high_freq_weight = if high_freq_support &gt; total_functions / 2 { 0.1 } else { 0.5 };
        let low_freq_weight = if low_freq_support &gt; total_functions / 2 { 0.1 } else { 0.5 };

        assert_eq!(high_freq_weight, 0.1, &quot;High frequency patterns should get low weight&quot;);
        assert_eq!(low_freq_weight, 0.5, &quot;Low frequency patterns should get higher weight&quot;);
    }

    /// Test percentile-based pattern selection
    #[test]
    fn test_percentile_pattern_selection() {
        // Create patterns with different support frequencies
        let patterns = vec![
            (&quot;very_common&quot;, 900),
            (&quot;common&quot;, 600),
            (&quot;medium&quot;, 300),
            (&quot;rare&quot;, 100),
            (&quot;very_rare&quot;, 10),
        ];

        let total_functions = 1000;
        let top_percentile_threshold = 0.95; // Top 5% most frequent

        // Calculate which patterns fall into top percentile
        let mut sorted_patterns = patterns.clone();
        sorted_patterns.sort_by(|a, b| b.1.cmp(&amp;a.1)); // Sort by support descending

        let top_5_percent_count = (patterns.len() as f64 * (1.0 - top_percentile_threshold)).ceil() as usize;
        let top_patterns: Vec&lt;_&gt; = sorted_patterns.iter().take(top_5_percent_count).collect();

        // Should include very_common pattern
        assert!(top_patterns.iter().any(|(name, _)| *name == &quot;very_common&quot;),
               &quot;Top percentile should include very common patterns&quot;);

        // Should not include very_rare pattern
        assert!(!top_patterns.iter().any(|(name, _)| *name == &quot;very_rare&quot;),
               &quot;Top percentile should not include very rare patterns&quot;);

        // Calculate contribution of top patterns
        let top_contribution: usize = top_patterns.iter().map(|(_, support)| *support).sum();
        let total_contribution: usize = patterns.iter().map(|(_, support)| *support).sum();
        let contribution_percentage = (top_contribution as f64 / total_contribution as f64) * 100.0;

        assert!(contribution_percentage &gt; 50.0, 
               &quot;Top percentile should contribute &gt;50% of total patterns: {}%&quot;, contribution_percentage);
    }
}

        let refresh_policy = CacheRefreshPolicy {

                    name: &quot;test_func&quot;.to_string(),
                    start_line: 1,
                    end_line: 10,
                    complexity: 5,
                }
            ],
        });
        
        let codebase_info = CodebaseInfo {
            functions: vec![
                FunctionInfo {
                    id: &quot;test_func_1&quot;.to_string(),
                    name: &quot;test_func&quot;.to_string(),

                    end_line: 10,
                    complexity: 5,
                }

            total_lines: 100,
            file_info,
        };

        let signature1 = manager.generate_codebase_signature(&amp;codebase_info);
        let signature2 = manager.generate_codebase_signature(&amp;codebase_info);

    /// Test multi-language cache with different pattern frequencies
    #[test]
    fn test_multi_language_cache() {
        let mut cache = StopMotifCache {
            version: 1,
            k_gram_size: 3,
            token_grams: vec![],
            pdg_motifs: vec![],
            ast_patterns: vec![],
            last_updated: chrono::Utc::now().timestamp() as u64,
            codebase_signature: &quot;multi_lang_test&quot;.to_string(),
            mining_stats: MiningStats {
                total_functions_analyzed: 1000,
                total_patterns_found: 5000,
                patterns_above_threshold: 500,
                top_1_percent_contribution: 20.0,
                processing_time_ms: 10000,
            },
        };

        // Add patterns from different languages
        let multi_lang_patterns = vec![
            // Python patterns (high frequency)
            (&quot;import os&quot;, 200, &quot;python&quot;, AstPatternCategory::Import),
            (&quot;def function&quot;, 180, &quot;python&quot;, AstPatternCategory::FunctionDeclaration),
            (&quot;if condition&quot;, 160, &quot;python&quot;, AstPatternCategory::ControlFlow),
            
            // JavaScript patterns (medium frequency)
            (&quot;console.log&quot;, 100, &quot;javascript&quot;, AstPatternCategory::FunctionCall),
            (&quot;function declaration&quot;, 90, &quot;javascript&quot;, AstPatternCategory::FunctionDeclaration),
            (&quot;const variable&quot;, 85, &quot;javascript&quot;, AstPatternCategory::VariableDeclaration),
            
            // TypeScript patterns (lower frequency)
            (&quot;interface definition&quot;, 50, &quot;typescript&quot;, AstPatternCategory::InterfaceDeclaration),
            (&quot;type annotation&quot;, 45, &quot;typescript&quot;, AstPatternCategory::TypeAnnotation),
            
            // Rust patterns (low frequency)
            (&quot;use crate&quot;, 30, &quot;rust&quot;, AstPatternCategory::Import),
            (&quot;fn function&quot;, 25, &quot;rust&quot;, AstPatternCategory::FunctionDeclaration),
            
            // Go patterns (low frequency)
            (&quot;package main&quot;, 20, &quot;go&quot;, AstPatternCategory::PackageDeclaration),
            (&quot;func main&quot;, 18, &quot;go&quot;, AstPatternCategory::FunctionDeclaration),
        ];

        for (pattern, support, language, category) in multi_lang_patterns {
            let idf_score = ((1.0 + cache.mining_stats.total_functions_analyzed as f64) / (1.0 + support as f64)).ln() + 1.0;
            let weight_multiplier = if support &gt; 100 { 0.1 } else { 0.3 };

            cache.ast_patterns.push(AstStopMotifEntry {
                pattern: pattern.to_string(),
                support,
                idf_score,
                weight_multiplier,
                category,
                language: language.to_string(),
                metadata: HashMap::new(),
            });
        }

        // Verify multi-language support
        let languages: HashSet&lt;String&gt; = cache.ast_patterns.iter()
            .map(|p| p.language.clone())
            .collect();
        assert_eq!(languages.len(), 5, &quot;Should support 5 different languages&quot;);
        assert!(languages.contains(&quot;python&quot;));
        assert!(languages.contains(&quot;javascript&quot;));
        assert!(languages.contains(&quot;typescript&quot;));
        assert!(languages.contains(&quot;rust&quot;));
        assert!(languages.contains(&quot;go&quot;));

        // Verify frequency-based IDF scoring
        let python_import = cache.ast_patterns.iter()
            .find(|p| p.pattern == &quot;import os&quot; &amp;&amp; p.language == &quot;python&quot;)
            .unwrap();
        let rust_import = cache.ast_patterns.iter()
            .find(|p| p.pattern == &quot;use crate&quot; &amp;&amp; p.language == &quot;rust&quot;)
            .unwrap();

        assert!(rust_import.idf_score &gt; python_import.idf_score,
               &quot;Lower frequency patterns should have higher IDF: Rust={}, Python={}&quot;, 
               rust_import.idf_score, python_import.idf_score);

        // Verify weight assignment based on frequency
        assert_eq!(python_import.weight_multiplier, 0.1, &quot;High frequency patterns should get low weight&quot;);
        assert_eq!(rust_import.weight_multiplier, 0.3, &quot;Low frequency patterns should get higher weight&quot;);
    }

    /// Test cache refresh triggers and policies
    #[test] 
    fn test_cache_refresh_policies() {
        let temp_dir = TempDir::new().unwrap();
        let cache_dir = temp_dir.path().to_path_buf();

        // Test different refresh policies
        let policies = vec![
            // Conservative policy
            CacheRefreshPolicy {
                auto_refresh_enabled: true,
                max_age_hours: 168, // 1 week
                min_codebase_change_threshold: 0.2, // 20% change required
                force_refresh_on_new_languages: false,
            },
            // Aggressive policy
            CacheRefreshPolicy {
                auto_refresh_enabled: true,
                max_age_hours: 6, // 6 hours
                min_codebase_change_threshold: 0.05, // 5% change required
                force_refresh_on_new_languages: true,
            },
            // Disabled policy
            CacheRefreshPolicy {
                auto_refresh_enabled: false,
                max_age_hours: 0,
                min_codebase_change_threshold: 1.0, // Never refresh
                force_refresh_on_new_languages: false,
            },
        ];

        for (i, policy) in policies.iter().enumerate() {
            let manager = StopMotifCacheManager::new(cache_dir.clone(), policy.clone());

            let old_cache = StopMotifCache {
                version: 1,
                k_gram_size: 3,
                token_grams: vec![],
                pdg_motifs: vec![],
                ast_patterns: vec![],
                last_updated: if policy.max_age_hours &gt; 0 {
                    chrono::Utc::now().timestamp() as u64 - (policy.max_age_hours as u64 * 3600 + 1)
                } else {
                    chrono::Utc::now().timestamp() as u64
                },
                codebase_signature: &quot;old_signature&quot;.to_string(),
                mining_stats: MiningStats {
                    total_functions_analyzed: 100,
                    total_patterns_found: 200,
                    patterns_above_threshold: 50,
                    top_1_percent_contribution: 10.0,
                    processing_time_ms: 1000,
                },
            };

            let should_refresh = manager.should_refresh_cache(&amp;old_cache, &quot;new_signature&quot;);

            match i {
                0 =&gt; { // Conservative
                    // Should refresh due to signature change (if auto refresh enabled)
                    assert_eq!(should_refresh, policy.auto_refresh_enabled);
                },
                1 =&gt; { // Aggressive
                    // Should refresh due to age and signature change
                    assert!(should_refresh, &quot;Aggressive policy should trigger refresh&quot;);
                },
                2 =&gt; { // Disabled
                    // Should not refresh when disabled
                    assert!(!should_refresh, &quot;Disabled policy should not trigger refresh&quot;);
                },
                _ =&gt; {}
            }
        }
    }

//! Test to validate token and signature caching effectiveness

use std::time::Instant;
use valknut_rs::core::featureset::CodeEntity;
use valknut_rs::detectors::lsh::LshExtractor;
use valknut_rs::core::config::LshConfig;

#[test]
fn test_caching_effectiveness() {
    let lsh_extractor = LshExtractor::new()
        .with_lsh_config(LshConfig {
            num_hashes: 64,
            num_bands: 8,
            shingle_size: 3,
            similarity_threshold: 0.7,
            max_candidates: 50,
            use_semantic_similarity: false,
        });
    
    // Create test source code
    let source_code = r#&quot;
        def calculate_fibonacci(n):
            if n &lt;= 1:
                return n
            else:
                return calculate_fibonacci(n-1) + calculate_fibonacci(n-2)
        
        def main():
            result = calculate_fibonacci(10)
            print(f&quot;Fibonacci of 10 is: {result}&quot;)
            return result
    &quot;#;
    
    println!(&quot;=== Testing Signature Caching Effectiveness ===&quot;);
    
    // Measure first signature generation (cache miss)
    let start_time = Instant::now();
    let signature1 = lsh_extractor.generate_minhash_signature(source_code);
    let first_time = start_time.elapsed();
    
    // Measure second signature generation (should be cache hit)
    let start_time = Instant::now();
    let signature2 = lsh_extractor.generate_minhash_signature(source_code);
    let second_time = start_time.elapsed();
    
    // Measure third signature generation (should be cache hit)
    let start_time = Instant::now();
    let signature3 = lsh_extractor.generate_minhash_signature(source_code);
    let third_time = start_time.elapsed();
    
    println!(&quot;First generation (cache miss): {:?}&quot;, first_time);
    println!(&quot;Second generation (cache hit): {:?}&quot;, second_time);
    println!(&quot;Third generation (cache hit): {:?}&quot;, third_time);
    
    // Verify signatures are identical
    assert_eq!(signature1, signature2, &quot;Cached signature should match original&quot;);
    assert_eq!(signature2, signature3, &quot;All cached signatures should match&quot;);
    
    // Cache hits should be significantly faster than cache miss
    // (We expect at least 50% speed improvement for cache hits)
    let cache_speedup = first_time.as_nanos() as f64 / second_time.as_nanos() as f64;
    println!(&quot;Cache speedup factor: {:.2}x&quot;, cache_speedup);
    
    // Get cache statistics
    let cache_stats = lsh_extractor.get_cache_statistics();
    println!(&quot;Cache statistics: {:?}&quot;, cache_stats);
    
    // Validate cache behavior
    assert!(cache_stats.signature_hits &gt;= 2, &quot;Should have at least 2 signature cache hits&quot;);
    // Note: token hits may be 0 if signature cache prevents token generation
    assert!(cache_stats.token_hits &gt;= 0, &quot;Token hits should be non-negative&quot;);
    
    let hit_rate = cache_stats.overall_hit_rate();
    println!(&quot;Overall cache hit rate: {:.1}%&quot;, hit_rate * 100.0);
    
    // With 3 calls to same source, we expect good hit rate after first miss
    assert!(hit_rate &gt; 0.0, &quot;Should have some cache hits&quot;);
}

fn test_cache_memory_management() {
    let lsh_extractor = LshExtractor::new();
    
    println!(&quot;\n=== Testing Cache Memory Management ===&quot;);
    
    // Generate many different source codes to test cache eviction
    let mut sources = Vec::new();
    for i in 0..100 {
        let source = format!(
            r#&quot;
            def function_{}():
                x = {}
                y = x * 2
                return y + {}
            &quot;#,
            i, i % 10, i % 5
        );
        sources.push(source);
    }
    
    println!(&quot;Generating signatures for {} different sources&quot;, sources.len());
    
    // Process all sources
    let start_time = Instant::now();
    for (i, source) in sources.iter().enumerate() {
        let _signature = lsh_extractor.generate_minhash_signature(source);
        
        if i % 20 == 0 {
            let cache_stats = lsh_extractor.get_cache_statistics();
            println!(&quot;After {} sources - hits: {}, misses: {}, evictions: {}&quot;, 
                     i + 1, 
                     cache_stats.signature_hits + cache_stats.token_hits,
                     cache_stats.signature_misses + cache_stats.token_misses,
                     cache_stats.evictions);
        }
    }
    
    let elapsed = start_time.elapsed();
    println!(&quot;Processed {} sources in {:?}&quot;, sources.len(), elapsed);
    
    // Now test cache effectiveness with repeated access
    let repeat_start = Instant::now();
    for source in sources.iter().take(10) {
        let _signature = lsh_extractor.generate_minhash_signature(source);
    }
    let repeat_elapsed = repeat_start.elapsed();
    
    println!(&quot;Repeated access to 10 sources: {:?}&quot;, repeat_elapsed);
    
    let final_stats = lsh_extractor.get_cache_statistics();
    println!(&quot;Final statistics: {:?}&quot;, final_stats);
    
    // Validate cache is working and managing memory
    assert!(final_stats.signature_hits &gt; 0 || final_stats.token_hits &gt; 0, 
            &quot;Should have some cache hits&quot;);
    
    let avg_time_per_source = elapsed.as_millis() / sources.len() as u128;
    println!(&quot;Average time per source: {}ms&quot;, avg_time_per_source);
    
    // Performance should be reasonable even with cache management
    assert!(avg_time_per_source &lt; 100, &quot;Average processing time should be reasonable&quot;);
